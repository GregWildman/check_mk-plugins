  IcomLicType ::= TEXTUAL-CONVENTION
   STATUS current
   DESCRIPTION 
         " the possible types of licenses provided.
           Normal - A full license without expiration time, produced for a specific mac address
           Demo - A license that expires after a certain time period (Up to 3 years)
           passpartout - A generic license designed to apply to many systems, independent to the
                        mac address. It is supposed to be replaced at some point with a normal one.
           nolicense - No license has been applied to the system, i.e. the factory default settings
                       are applied"
   SYNTAX INTEGER { 
      normal(1),
      demo(2),
      passpartout(3),
      nolicense(4),
      demopassepartout(5)
      }

   IcomLicApplyTypes ::= TEXTUAL-CONVENTION
   STATUS current
   DESCRIPTION 
         " Types of license regarding the application level:
           Standard - The license is full (all features) and forced, without any checks
           Upgrade - The license may be full or partial (only some features) and checks
                     are performed before application"
   SYNTAX INTEGER {   
      standard(1),
      upgrade(2)
      }

   icomLicActExpStatusNotification NOTIFICATION-TYPE
      OBJECTS {icomLicActExpirationStatus}
      STATUS current
      DESCRIPTION "
            A trap indicating License expiration status
           [OmniBAS-2W, OmniBAS-4W  implementation]
            The NMS forwards the following information to the operator:
               varbind[0].oid = oid_sysUpTime.0 = 1.3.6.1.2.1.1.3.0
               varbind[0].val = NMS agent time (in centiseconds)
           
               varbind[1].oid = oid_snmpTrapOID.0 = 1.3.6.1.6.3.1.1.4.1.0
               varbind[1].val = icomLicActExpStatusNotification
                              = 1.3.6.1.4.1.1807.101.0.1
               varbind[2].oid = icomLicActExpirationStatus
                              = 1.3.6.1.4.1.1807.101.1.2.1.17
               varbind[2].val = icomLicActExpirationStatus value
           
             where
           -- 
                icomLicActExpirationStatus is the license server 
                received license expiration indication.
                The information is TruthValue with the following meaning:
                      - 2(false): License is not expired
                      - 1(true) : License is expired
                            
      "
         ::= { icomLicMIBNotifObjects 1}
   
   icomLicActProtConflictNotification NOTIFICATION-TYPE
      OBJECTS {icomLicActNetworkRedundancySupport}
      STATUS current
      DESCRIPTION "
            A trap indicating a conflict caused by the value of network redundancy feature in active license.
            This can be caused in systems supporting protection when
                - There is only one card, so that protection has no meaning
                - There are two cards but the communication between them is not feasible
                - There are two cards but their licenses differ.
            The NMS forwards the following information to the operator:
               varbind[0].oid = oid_sysUpTime.0 = 1.3.6.1.2.1.1.3.0
               varbind[0].val = NMS agent time (in centiseconds)
           
               varbind[1].oid = oid_snmpTrapOID.0 = 1.3.6.1.6.3.1.1.4.1.0
               varbind[1].val = icomLicActProtConflictNotification
                              = 1.3.6.1.4.1.1807.101.0.2
               varbind[2].oid = icomLicActNetworkRedundancySupport
                              = 1.3.6.1.4.1.1807.101.1.2.1.14
               varbind[2].val = icomLicActNetworkRedundancySupport value
           
             where
           -- 
                icomLicActNetworkRedundancySupport is the license server 
                received active license protection (network redundancy) feature value.
                The information is TruthValue with the following meaning:
                      - 2(false): Protection is not supported
                      - 1(true) : Protection is supported
                            
      "
         ::= { icomLicMIBNotifObjects 2}
         
   
   icomLicActOpModeConflictNotification NOTIFICATION-TYPE
      OBJECTS {icomLicActOperationMode}
      STATUS current
      DESCRIPTION "
            A trap indicating a conflict caused by the value of Operation feature in upgrade license.
            This conflict can be caused when the featured Operation Mode value is not compatible 
            with the value of system type in the active license.
            The NMS forwards the following information to the operator:
               varbind[0].oid = oid_sysUpTime.0 = 1.3.6.1.2.1.1.3.0
               varbind[0].val = NMS agent time (in centiseconds)
           
               varbind[1].oid = oid_snmpTrapOID.0 = 1.3.6.1.6.3.1.1.4.1.0
               varbind[1].val = icomLicActOpModeConflictNotification
                              = 1.3.6.1.4.1.1807.101.0.3
               varbind[2].oid = icomLicUpgrOperationMode
                              = 1.3.6.1.4.1.1807.101.1.5.1.12
               varbind[2].val = icomLicUpgrOperationMode value
               varbind[3].oid = icomLicActSystemType
                              = 1.3.6.1.4.1.1807.101.1.2.1.2
               varbind[3].val = icomLicActSystemType value
             where
           -- 
                icomLicUpgrOperationMode is the license server 
                received upgrade license operation mode feature value.
                The information is IcomLicOpModeBitmap with the following meaning:
                Bit0:
                      - 0: Ptp Mode not supported
                      - 1: Ptp Mode supported
                Bit1:
                      - 0: Hub Mode not supported
                      - 1: Hub Mode supported
                Bit2:
                      - 0: Terminal Mode not supported
                      - 1: Terminal Mode supported
                Both Hub and terminal modes refer to PmP.
                
                icomLicActSystemType is the System type of the system as kept in active license.
                Its value is one of these declared in IcomLicProductTypeValue TC.
                            
      "
         ::= { icomLicMIBNotifObjects 3}
   
   icomLicActExpWarning NOTIFICATION-TYPE
      OBJECTS {icomLicActDuration}
      STATUS current
      DESCRIPTION "
            A trap indicating the time (in days) remaining till the license expiration 
            when this time becomes less than a fixed number of days.
            The NMS forwards the following information to the operator:
               varbind[0].oid = oid_sysUpTime.0 = 1.3.6.1.2.1.1.3.0
               varbind[0].val = NMS agent time (in centiseconds)
           
               varbind[1].oid = oid_snmpTrapOID.0 = 1.3.6.1.6.3.1.1.4.1.0
               varbind[1].val = icomLicActExpWarning
                              = 1.3.6.1.4.1.1807.101.0.4
               varbind[2].oid = icomLicActDuration
                              = 1.3.6.1.4.1.1807.101.1.2.1.16
               varbind[2].val = icomLicActDuration value
             where
           -- 
               icomLicActDuration is the active license number of days, until expiration. Maximum duration is 3 years.
               The value 0 could be used for non-demo license, that does not actually expire.                
                            
      "
         ::= { icomLicMIBNotifObjects 4}
